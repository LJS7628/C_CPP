{
	int i, j, min, index, temp,k;
	int array[11] = {};

	for(int e=0; e<11; e++)
	{
	 int array[e]=rand()%100;
	}
	for (i = 0; i < 11; i++)
	{
		min = 9999;
		for (j = i; j < 11; j++)
		{
			if (min > array[j])

				min = array[j];

			}
		}
		temp = array[i];

			printf("%2d ", array[k]);			
		}

	}

}
