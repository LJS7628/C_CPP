#include "stdafx.h"


int main()
{
	int i, j, k;

	for (i = 0; i < 9; i++)
	{
		for (j = 0; j <9-i; j++)
		{
			printf(" ");
		}
		for (k = 0; k < i+1; k++)
		{
			printf("*");
		}
		printf("\n");
	}
	
	for (i = 0; i < 8; i++)
	{
		for (j = 0; j < i + 2; j++)
		{
			printf(" ");
		}
		for (k=0;k<8-i;k++)
		{
			printf("*");
		}
		printf("\n");
	}
	return 0;
}
