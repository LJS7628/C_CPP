#include "stdafx.h"
#include <stdlib.h>
#include <time.h>

int main()
{
	int i, j, temp,k;
	int array[11] = {};
	srand((unsigned)time(NULL));
	for (int e = 0; e < 11; e++)
	{
		array[e] = rand() % 100;
	}
	for (i = 0; i < 10; i++)
	{
		for (j = 0; j < 9 - i; j++)
		{
			if (array[j] > array[j + 1])
			{
				temp = array[j];
				array[j] = array[j + 1];
				array[j + 1] = temp;
			}
			for (k = 0; k < 10; k++)
			{
				printf("%d ", array[k]);
			}
			printf("\n\n");
		}
		
	}

	
	return 0;
}
